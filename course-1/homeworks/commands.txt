git init — создаёт новый репозиторий
git add — добавляет файлы в индекс
git branch - показать список всех веток
git branch [имя ветки] - создать новую ветку
git checkout [имя ветки] - переключиться на указанную ветку
git checkout -b [имя ветки] - создать и перейти в ветку
git merge [имя ветки] - слияние текущей ветки с указанной
